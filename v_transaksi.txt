SELECT td.*,th.*,j.*,b.*,jo.*,kb.*,u.*,ka.*,p.*,ka_2.ka_akun as pasien_ka FROM transaksi_detail td LEFT JOIN transaksi_head th ON td.td_head=th.th_kode LEFT JOIN jasa j ON td.td_jasa=j.jasa_id LEFT JOIN barang b ON td.td_barang=b.barang_id LEFT JOIN jenis_obat jo ON b.barang_jenis=jo.jo_id LEFT JOIN kelompok_barang kb ON b.barang_kelompok=kb.kb_id LEFT JOIN user u ON td.td_update_user=u.user_id LEFT JOIN kode_akun ka ON th.th_kode_akuntansi=ka.ka_id LEFT JOIN pasien p ON th.th_pasien=p.pasien_id LEFT JOIN kode_akun ka_2 ON p.pasien_tipe=ka_2.ka_id
